```sql
CREATE TABLE `user` (
  `id` int(10) unsigned NOT NULL AUTO_INCREMENT COMMENT '用户ID',
  `third_account_id` varchar(128) NOT NULL DEFAULT '' COMMENT '第三方账号ID',
  `user_name` varchar(64) NOT NULL DEFAULT '' COMMENT '用户名',
  `password` varchar(128) NOT NULL DEFAULT '' COMMENT '密码',
  `login_type` tinyint(1) NOT NULL DEFAULT '0' COMMENT '登录方式，0-账号密码登录，1-微信登录',
  `is_deleted` tinyint(1) NOT NULL DEFAULT '0' COMMENT '是否删除，0-未删除，1-已删除',
  `create_time` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '最后更新时间',
  PRIMARY KEY (`id`),
  KEY `key_third_account_id` (`third_account_id`),
  KEY `key_user_name` (`user_name`) 
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8mb4 COMMENT='用户登录表';
```

```sql
CREATE TABLE `user_info` (
  `id` int(10) unsigned NOT NULL AUTO_INCREMENT COMMENT '用户信息ID',
  `user_id` int(10) unsigned NOT NULL DEFAULT '0' COMMENT '用户ID',
  `nick_name` varchar(64) NOT NULL DEFAULT '' COMMENT '昵称',
  `avatar_url` varchar(256) NOT NULL DEFAULT '' COMMENT '头像URL',
  `gender` tinyint(1) NOT NULL DEFAULT '0' COMMENT '性别，0-未知，1-男，2-女',
  `position` varchar(50) NOT NULL DEFAULT '' COMMENT '职位',
  `company` varchar(50) NOTNULL DEFAULT '' COMMENT '公司',
  `profile` varchar(256) NOT NULL DEFAULT '' COMMENT '个人简介',
  `ip` varchar(64) NOT NULL DEFAULT '' COMMENT 'IP地址',
  `birthday` date DEFAULT NULL COMMENT '生日',
  `signature` varchar(256) NOT NULL DEFAULT '' COMMENT '个性签名',
  `extend` varchar(1024) NOT NULL DEFAULT '' COMMENT '扩展字段',
  `is_deleted` tinyint(1) NOT NULL DEFAULT '0' COMMENT '是否删除，0-未删除，1-已删除',
  `create_time` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '最后更新时间',
  PRIMARY KEY (`id`),
  KEY `key_user_id` (`user_id`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8mb4 COMMENT='用户个人信息表';
```

```sql
CREATE TABLE `user_relation` (
  `id` int(10) unsigned NOT NULL AUTO_INCREMENT COMMENT '用户关系ID',
  `user_id` int(10) unsigned NOT NULL DEFAULT '0' COMMENT '作者用户ID',
  `follow_user_id` int(10) unsigned NOT NULL DEFAULT '0' COMMENT '关注的userID的用户ID，即粉丝ID',
  `follow_state` tinyint(1) NOT NULL DEFAULT '0' COMMENT '关注状态，0-未关注，1-已关注，2-取消关注',
  `is_deleted` tinyint(1) NOT NULL DEFAULT '0' COMMENT '是否删除，0-未删除，1-已删除',
  `create_time` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '最后更新时间',
  PRIMARY KEY (`id`),
  KEY `key_user_id` (`user_id`),
  UNIQUE KEY `uk_user_follow` (`user_id`,`follow_user_id`),
  KEY `key_follow_user_id` (`follow_user_id`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8mb4 COMMENT='用户关系表';
```
